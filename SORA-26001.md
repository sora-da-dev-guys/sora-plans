### API Pattern (follow exactly from `apis.ts`)

```ts
export const getSaleOrderTableDataApi = async (
  query: GetSaleOrderTableDataQuery
): Promise<ISaleOrderTableData> => {
  const queryString = buildQueryString(query as Record<string, string | undefined>)
  const response = await fetch(`${SBB_API_URL}/sale-orders/table-data${queryString}`, {
    method: 'GET',
    headers: getAuthHeaders(),
  })
  if (!response.ok) {
    if (response.status === 401) handleUnauthorized()
    const error = await response.json().catch(() => ({}))
    throw new Error((error as { message?: string }).message || 'Failed to fetch sale order table data')
  }
  return response.json()
}

export const getUserListApi = async (): Promise<IUserListData[]> => {
  const response = await fetch(`${SBB_API_URL}/data/v1/user-data`, {

    throw new Error((error as { message?: string }).message || 'Failed to fetch user list')
  }
  return response.json()
}
```

### State Management

All state via `useSearchParams` (follows CustomerListPage pattern):

| State | Search Param |
|-------|-------------|
| Active tab | `?tab=0` (0-6) |
| Page | `?page=0` |
| Search inputs | `?code=`, `?customerName=`, etc. |
| Filters | `?statusList=`, `?paymentStatusList=`, etc. |
| Sort | `?sortKey=0` |

- Debounce all text search inputs at 500ms (same as CustomerListPage)
- Reset `page` to `0` when any filter/search changes
